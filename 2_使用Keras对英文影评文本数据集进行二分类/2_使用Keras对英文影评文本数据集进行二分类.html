word2id = imdb.get_word_index()
word2id = {w: i + 3 for w, i in word2id.items()} # id从1开始
word2id['<PAD>'] = 0                             # 填充字符
word2id['<START>'] = 1                           # 句子的开始
word2id['<UNK>'] = 2                             # 词典以外的词
word2id['<UNUSED>'] = 3                          # 预留

id2word = {i: w for w, i in word2id.items()}     # id到词的映射

# 将id序列变为原始文本的函数
def decode_review(ids):                  
    return ' '.join([id2word.get(i, '?') for i in ids])

print(decode_review(X_train[0]))

定义模型并编译，从训练集中分割出一部分作为验证集（Validation Set），使用训练集和验证集训练模型

  • Embedding对id序列进行嵌入，将每个id表示为向量，这样一来25000*256变成了25000*256*16
  • GlobalAveragePooling1D对序列长度这一维度进行全局平均池化，这样一来25000*256*16变成了25000*16
  • 多分类最后一层Dense一般用softmax激活函数，二分类则用sigmoid
  • 二分类的损失函数一般用binary_crossentropy
  • batch_size为训练的批大小，即每次使用若干条数据进行训练，如不提供则为32
In [5]:
# 定义模型
model = keras.Sequential()
model.add(keras.layers.Embedding(vocab_size, 16))
model.add(keras.layers.GlobalAveragePooling1D())
model.add(keras.layers.Dense(16, activation='relu'))
model.add(keras.layers.Dense(1, activation='sigmoid'))
model.summary()

# 编译模型
model.compile(optimizer='adam', loss='binary_crossentropy', metrics=['accuracy'])

# 训练集和验证集分割
train_size = 20000
X_val = X_train[train_size:]
X_train = X_train[:train_size]
Y_val = Y_train[train_size:]
Y_train = Y_train[:train_size]

# 训练模型
history = model.fit(X_train, Y_train, epochs=40, batch_size=512, validation_data=(X_val, Y_val))
